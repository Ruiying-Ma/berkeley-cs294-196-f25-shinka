<DESCRIPTION>
Refine W-TinyLFU implementation with:
1.  **Doorkeeper & Frequency Sketch**: Use a Doorkeeper (set) to filter one-hit wonders and a frequency map capped at 15 to track item popularity.
2.  **Adaptive Aging**: Halve frequencies every 5x capacity accesses to adapt to workload phase changes.
3.  **Correct Window LRU**: Switch 'window' from FIFO deque to OrderedDict LRU to properly track recency within the window.
4.  **Biased Duel**: When the window is full, duel the Window LRU candidate against the Main LRU candidate. Only evict Main if the Window candidate has a strictly higher frequency.
5.  **Robust State Management**: Ensure state resets correctly between traces and handle cache sizing dynamically.

# W-TinyLFU style state
algo_state = {
    'window': deque(),        # Queue of (key, size)
    'window_size': 0,         # Current size of window
    'probation': OrderedDict(), # key -> size
    'protected': OrderedDict(), # key -> size
    'protected_size': 0,
    'freq': {},               # key -> count
    'access_count': 0
}

def evict(cache_snapshot, obj):
    '''
    W-TinyLFU Eviction:
    - Maintains a small Window (1%) and a Main SLRU (Probation + Protected).
    - If Window is not full, evict from Main (to allow Window to grow).
    - If Window is full, duel the Window victim vs Main victim using frequency.
    '''
    window = algo_state['window']
    probation = algo_state['probation']
    protected = algo_state['protected']
    freq = algo_state['freq']

    capacity = cache_snapshot.capacity
    w_cap = max(1, int(capacity * 0.01))

    # 1. If Window has space, prefer evicting from Main (Probation -> Protected)
    if algo_state['window_size'] < w_cap:
        if probation:
            return next(iter(probation))
        if protected:
            return next(iter(protected))
        if window:
            return window[0][0]

    # 2. Window is full. Find candidates.
    if not window:
        # Fallback if window empty (rare)
        if probation: return next(iter(probation))
        if protected: return next(iter(protected))
        return None

    cand_w_key = window[0][0]

    if probation:
        cand_m_key = next(iter(probation))
    elif protected:
        cand_m_key = next(iter(protected))
    else:
        return cand_w_key

    # Duel: Keep the one with higher frequency
    freq_w = freq.get(cand_w_key, 0)
    freq_m = freq.get(cand_m_key, 0)

    if freq_w > freq_m:
        return cand_m_key
    else:
        return cand_w_key

def update_after_hit(cache_snapshot, obj):
    '''
    Update frequency and manage SLRU promotion.
    '''
    key = obj.key
    algo_state['access_count'] += 1

    # Frequency Update & Aging
    f = algo_state['freq']
    f[key] = f.get(key, 0) + 1
    if algo_state['access_count'] > cache_snapshot.capacity * 10:
        algo_state['freq'] = {k: v // 2 for k, v in f.items() if v > 1}
        algo_state['access_count'] = 0

    # SLRU Management
    if key in algo_state['protected']:
        algo_state['protected'].move_to_end(key)
    elif key in algo_state['probation']:
        # Promote from Probation to Protected
        val = algo_state['probation'].pop(key)
        algo_state['protected'][key] = val
        algo_state['protected_size'] += val

        # Enforce Protected Limit (80% of capacity)
        limit = int(cache_snapshot.capacity * 0.8)
        while algo_state['protected_size'] > limit and algo_state['protected']:
            k, v = algo_state['protected'].popitem(last=False)
            algo_state['protected_size'] -= v
            algo_state['probation'][k] = v

def update_after_insert(cache_snapshot, obj):
    '''
    Insert new object into Window and handle overflow to Probation.
    '''
    key = obj.key
    size = obj.size
    algo_state['access_count'] += 1

    # Frequency Update
    f = algo_state['freq']
    f[key] = f.get(key, 0) + 1

    # Insert into Window
    algo_state['window'].append((key, size))
    algo_state['window_size'] += size

    # Check Window Overflow -> Move to Probation
    w_cap = max(1, int(cache_snapshot.capacity * 0.01))

        k, s = algo_state['window'].popleft()
        algo_state['window_size'] -= s
        algo_state['probation'][k] = s

algo_state = {
    'window': OrderedDict(),    # key -> size (LRU)
    'window_size': 0,
    'probation': OrderedDict(), # key -> size (LRU)
    'protected': OrderedDict(), # key -> size (LRU)
    'protected_size': 0,
    'freq': {},                 # key -> count
    'doorkeeper': set(),        # key (Bloom Filter approximation)
    'access_count': 0
}

def _check_reset(cache_snapshot):
    # Reset state if cache is empty (new trace)
    if not cache_snapshot.cache and (algo_state['window'] or algo_state['probation'] or algo_state['protected']):
        algo_state['window'].clear()
        algo_state['window_size'] = 0
        algo_state['probation'].clear()
        algo_state['protected'].clear()
        algo_state['protected_size'] = 0
        algo_state['freq'].clear()
        algo_state['doorkeeper'].clear()
        algo_state['access_count'] = 0

def _get_freq(key):
    if key in algo_state['freq']:
        return algo_state['freq'][key]
    if key in algo_state['doorkeeper']:
        return 1
    return 0

def _record_access(key, capacity):
    algo_state['access_count'] += 1
    f = algo_state['freq']
    dk = algo_state['doorkeeper']

    # Cap frequency at 15
    if key in f:
        f[key] = min(f[key] + 1, 15)
    elif key in dk:
        dk.remove(key)
        f[key] = 2
    else:
        dk.add(key)

    # Aging
    interval = capacity * 5
    if algo_state['access_count'] >= interval:
        rem = []
        for k in f:
            f[k] //= 2
            if f[k] == 0: rem.append(k)
        for k in rem: del f[k]

        # Reset Doorkeeper if too large
        if len(dk) > capacity * 2:
            dk.clear()
        algo_state['access_count'] = 0

def evict(cache_snapshot, obj):
    '''
    W-TinyLFU with optimized parameters:
    - 1% Window, 99% Main (80% Protected, 20% Probation)
    - TinyLFU Admission with Doorkeeper
    '''
    _check_reset(cache_snapshot)

    window = algo_state['window']
    probation = algo_state['probation']
    protected = algo_state['protected']

    capacity = cache_snapshot.capacity
    w_cap = max(1, int(capacity * 0.01))

    # Candidates
    cand_w_key = next(iter(window)) if window else None

    cand_m_key = None
    if probation:
        cand_m_key = next(iter(probation))
    elif protected:
        cand_m_key = next(iter(protected))

    if not cand_m_key: return cand_w_key
    if not cand_w_key: return cand_m_key

    # 1. If Window not full, evict Main to grow Window
    if algo_state['window_size'] < w_cap:
        return cand_m_key

    # 2. Window Full -> Duel
    fw = _get_freq(cand_w_key)
    fm = _get_freq(cand_m_key)

    # Biased Duel: Evict Main only if Window is strictly better
    if fw > fm:
        return cand_m_key
    else:
        return cand_w_key
