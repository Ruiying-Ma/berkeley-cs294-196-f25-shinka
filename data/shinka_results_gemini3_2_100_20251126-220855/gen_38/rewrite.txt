"""Cache eviction algorithm for optimizing hit rates across multiple workloads"""
from collections import OrderedDict
import random

# ARC Global State with Frequency Augmentation
m_t1 = OrderedDict() # T1: Recent cache entries
m_t2 = OrderedDict() # T2: Frequent cache entries
m_b1 = OrderedDict() # B1: Ghost entries evicted from T1
m_b2 = OrderedDict() # B2: Ghost entries evicted from T2
m_freq = {}          # Global frequency map (persistent across evictions)
m_p = 0              # Target size for T1
m_last_access_count = -1
m_aggr_access = 0    # Counter for aging

def _check_reset(cache_snapshot):
    """Resets global state if a new trace is detected based on access count rollback."""
    global m_t1, m_t2, m_b1, m_b2, m_p, m_last_access_count, m_freq, m_aggr_access
    if cache_snapshot.access_count < m_last_access_count:
        m_t1.clear()
        m_t2.clear()
        m_b1.clear()
        m_b2.clear()
        m_freq.clear()
        m_p = 0
        m_aggr_access = 0
    m_last_access_count = cache_snapshot.access_count

    # Frequency Aging: Halve frequencies periodically to adapt to shifting workloads
    # Aging period set to 10 * capacity to provide history but prevent staleness
    m_aggr_access += 1
    if m_aggr_access > cache_snapshot.capacity * 10:
        for k in list(m_freq):
            m_freq[k] //= 2
            if m_freq[k] == 0:
                del m_freq[k]
        m_aggr_access = 0

def evict(cache_snapshot, obj):
    '''
    Smart ARC Eviction Policy:
    - Adapts 'p' (target T1 size) based on ghost hits.
    - Uses jittered 'p' to determine if T1 or T2 should be the victim source.
    - T1 Eviction: Selects a victim from the bottom k LRU items, minimizing global frequency.
      This preserves items that have historical significance but are currently in T1.
    - T2 Eviction: LRU.
    '''
    _check_reset(cache_snapshot)
    global m_p

    key = obj.key
    capacity = cache_snapshot.capacity

    # 1. Adapt p (target size of T1) based on hits in ghost lists
    # Note: If key is in ghost list, it indicates the cache was too small in that region.
    if key in m_b1:
        delta = 1
        if len(m_b1) > 0 and len(m_b2) > 0:
            delta = max(1, len(m_b2) / len(m_b1))
        m_p = min(capacity, m_p + delta)
    elif key in m_b2:
        delta = 1
        if len(m_b1) > 0 and len(m_b2) > 0:
            delta = max(1, len(m_b1) / len(m_b2))
        m_p = max(0, m_p - delta)

    # 2. Select victim source
    # Add jitter to target T1 size to avoid boundary resonance
    jitter = random.randint(-1, 1)
    target_t1 = max(0, min(capacity, m_p + jitter))

    evict_t1 = False
    if len(m_t1) > 0:
        if len(m_t1) > target_t1:
            evict_t1 = True
        elif key in m_b2 and len(m_t1) >= int(target_t1):
            evict_t1 = True

    # If T2 is empty, we must evict from T1
    if len(m_t2) == 0:
        evict_t1 = True

    victim_key = None

    if evict_t1 and m_t1:
        # Smart T1 Eviction:
        # Inspect bottom k items (LRU end).
        # Evict the one with lowest global frequency.
        # Break ties randomly to maintain scan resistance.
        k = 5
        candidates = []
        it = iter(m_t1)
        try:
            for _ in range(k):
                candidates.append(next(it))
        except StopIteration:
            pass
        
        # Select best victim: Min Frequency. Tie-breaker: Random.
        best_victim = None
        min_score = float('inf')
        
        for cand in candidates:
            # Score = freq + noise. 
            # Lower freq is better victim. Random noise breaks ties.
            f = m_freq.get(cand, 0)
            score = f + random.random()
            
            if score < min_score:
                min_score = score
                best_victim = cand
        
        return best_victim
    else:
        # Evict LRU from T2
        # T2 items are generally frequent. LRU is standard.
        return next(iter(m_t2)) if m_t2 else next(iter(m_t1))

def update_after_hit(cache_snapshot, obj):
    '''
    On Cache Hit:
    - Update Frequency.
    - Move to MRU of T2 (promote from T1 if needed).
    '''
    _check_reset(cache_snapshot)
    key = obj.key
    m_freq[key] = m_freq.get(key, 0) + 1

    if key in m_t1:
        del m_t1[key]
        m_t2[key] = 1
        m_t2.move_to_end(key)
    elif key in m_t2:
        m_t2.move_to_end(key)
    else:
        # Recovery/Self-healing
        m_t2[key] = 1
        m_t2.move_to_end(key)

def update_after_insert(cache_snapshot, obj):
    '''
    On Cache Insert (Miss):
    - Update Frequency.
    - If in B1 -> Promote to T2 (Restore).
    - If in B2 -> Promote to T2 (Restore).
    - Else -> Insert to T1 (New/Probation).
    '''
    _check_reset(cache_snapshot)
    key = obj.key
    m_freq[key] = m_freq.get(key, 0) + 1

    if key in m_b1:
        del m_b1[key]
        m_t2[key] = 1
        m_t2.move_to_end(key)
    elif key in m_b2:
        del m_b2[key]
        m_t2[key] = 1
        m_t2.move_to_end(key)
    else:
        # New item to T1
        m_t1[key] = 1
        m_t1.move_to_end(key)

def update_after_evict(cache_snapshot, obj, evicted_obj):
    '''
    On Eviction:
    - Move evicted key from T1/T2 to B1/B2 (Ghost lists).
    - Restrict ghost list size to capacity.
    '''
    key = evicted_obj.key
    capacity = cache_snapshot.capacity

    if key in m_t1:
        del m_t1[key]
        m_b1[key] = 1
        m_b1.move_to_end(key)
    elif key in m_t2:
        del m_t2[key]
        m_b2[key] = 1
        m_b2.move_to_end(key)

    # Restrict ghost size
    if len(m_b1) > capacity:
        m_b1.popitem(last=False)
    if len(m_b2) > capacity:
        m_b2.popitem(last=False)
