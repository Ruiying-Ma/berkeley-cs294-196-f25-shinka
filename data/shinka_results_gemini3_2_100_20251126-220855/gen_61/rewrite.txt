"""S3-FIFO with Extended Ghost Registry"""

# Global Metadata
# s3_small: FIFO queue for new items (probation), target size 10%
# s3_main: Segmented FIFO (Clock) for promoted items, target size 90%
# s3_ghost: Ghost cache tracking evicted small items to facilitate rescue
# s3_freq: Frequency counters (2-bit: 0-3) for items
s3_small = {}
s3_main = {}
s3_ghost = {}
s3_freq = {}
m_last_access_count = -1

def _check_reset(cache_snapshot):
    """Resets global state if a new trace is detected."""
    global s3_small, s3_main, s3_ghost, s3_freq, m_last_access_count
    if cache_snapshot.access_count < m_last_access_count:
        s3_small.clear()
        s3_main.clear()
        s3_ghost.clear()
        s3_freq.clear()
    m_last_access_count = cache_snapshot.access_count

def evict(cache_snapshot, obj):
    '''
    S3-FIFO Eviction Policy:
    1. Clean Ghost: Ensure ghost size <= 3x Capacity (Extended).
    2. Check Small: If > 10% capacity or Main is empty, evict from Small.
       - If freq > 0 in Small, promote to Main (reset freq).
       - Else, evict to Ghost.
    3. Check Main: If Small is within limits, evict from Main.
       - If freq > 0, decrement and reinsert (Clock).
       - Else, evict (drop).
    '''
    global s3_small, s3_main, s3_ghost, s3_freq
    _check_reset(cache_snapshot)

    capacity = cache_snapshot.capacity
    # Target size for Small queue (10% of capacity)
    s_capacity = max(1, int(capacity * 0.1))
    
    # Extended Ghost Capacity: 3x cache size
    # This helps capture loops larger than the cache size (improving Trace 14)
    g_capacity = int(capacity * 3)

    # Lazy cleanup of ghost
    while len(s3_ghost) > g_capacity:
        k = next(iter(s3_ghost))
        s3_ghost.pop(k)
        # Clean up frequency to prevent memory leak
        if k in s3_freq:
            del s3_freq[k]

    while True:
        # Decision: Evict from Small or Main?
        # Rule: Evict from Small if it exceeds its target size, OR if Main is empty.
        if len(s3_small) >= s_capacity or not s3_main:
            if not s3_small:
                # Should not happen if cache is full, but safety valve
                return None

            candidate = next(iter(s3_small))
            freq = s3_freq.get(candidate, 0)

            if freq > 0:
                # Hit in Small: Promote to Main
                # Remove from Small, Add to Main
                s3_small.pop(candidate)
                s3_main[candidate] = None
                # Reset frequency to 0 (Probation in Main)
                s3_freq[candidate] = 0
                continue
            else:
                # Victim found in Small (freq == 0)
                # This candidate will be evicted and moved to Ghost in update_after_evict
                return candidate

        else:
            # Evict from Main
            candidate = next(iter(s3_main))
            freq = s3_freq.get(candidate, 0)

            if freq > 0:
                # Hit in Main: Decrement frequency and reinsert at tail (Clock logic)
                s3_main.pop(candidate)
                s3_main[candidate] = None
                s3_freq[candidate] = freq - 1
                continue
            else:
                # Victim found in Main (freq == 0)
                # This candidate will be dropped
                return candidate

def update_after_hit(cache_snapshot, obj):
    '''
    On Hit: Increment frequency, capped at 3 (2-bit counter).
    '''
    _check_reset(cache_snapshot)
    key = obj.key
    s3_freq[key] = min(s3_freq.get(key, 0) + 1, 3)

def update_after_insert(cache_snapshot, obj):
    '''
    On Insert:
    - If in Ghost, rescue to Main (reset freq to 0).
    - Else, insert to Small (freq 0).
    '''
    _check_reset(cache_snapshot)
    global s3_small, s3_main, s3_ghost, s3_freq
    key = obj.key
    
    if key in s3_ghost:
        # Rescue: Ghost -> Main
        s3_main[key] = None
        s3_ghost.pop(key)
        # Reset freq to 0. It gets a full cycle in Main.
        s3_freq[key] = 0
    else:
        # Insert New: Small
        s3_small[key] = None
        s3_freq[key] = 0

def update_after_evict(cache_snapshot, obj, evicted_obj):
    '''
    On Evict:
    - If evicted from Small, add to Ghost.
    - If evicted from Main, just remove.
    '''
    global s3_small, s3_main, s3_ghost, s3_freq
    key = evicted_obj.key

    if key in s3_small:
        s3_small.pop(key)
        s3_ghost[key] = None
        # Keep s3_freq entry for Ghost items (it's 0, but valid)
    elif key in s3_main:
        s3_main.pop(key)
        # Remove freq for Main evictions (history lost)
        if key in s3_freq:
            del s3_freq[key]
